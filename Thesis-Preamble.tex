\fi

\ifdim\wd\@tempboxa>\hsize
{{\sffamily\bfseries{#1.}}\hspace{0.5em}#2}%
\else
\hb@xt@\hsize{\hfil\box\@tempboxa\hfil}%
\fi
\ifx\@captype\@tablestring
\vskip1em
\fi
}
